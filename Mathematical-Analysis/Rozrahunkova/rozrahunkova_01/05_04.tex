$$
  \begin{array}{r c l}
    y  & = & (x^3-x^2)^x \\
    y' & = & ((x^3-x^2)^x)' \\
    (\ln y)' & = & (\ln(x^3-x^2)^x)' \\
    \\
    \dfrac{1}{y}y' & = & x \dfrac{1}{x^3-x^2}(x^3-x^2)' \\
  \end{array}
$$

$$
  y' = x(x^3-x^2)^x \dfrac{1}{x^3-x^2}(3x^2-2x)
      = \dfrac{x (x^3-x^2)^x(3x^2-2x)}{x^3-x^2}
$$
$$
  y' = (x^3-x^2)^{x-1}(3x^3-2x^2).
$$
