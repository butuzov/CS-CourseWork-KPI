$$
  y = \dfrac{1}{2}\ln(e^{2x}+1)+\tg{e^x}
$$

$$
  y' = \dfrac{1}{2}(\ln(e^{2x}+1))'+(\tg{e^x})'
     = \ln(e^{2x}+1)\cdot(e^{2x}+1)' + \dfrac{1}{\cos e^x} (e^x)'
     = \ln(e^{2x}+1)\cdot(e^{2x}\cdot(2x)') + \dfrac{1}{\cos e^x} e^x
$$
$$
  y' = \ln(e^{2x}+1) 2e^{2x} + \frac{e^x}{\cos e^x}
$$
