{\descr{Обчислити визначений інтеграл}}

$$
  \int^{-\sqrt[3]{3}}_{-2} \dfrac{\d{x}}{x\sqrt{1-x^3}}
$$



% рішення за методом біному але не правильне

$\lhd I = x^{-1} \cdot (1-x^3)^{-1/2}  $ то $m=-1$, $n=3$, $p=-\dfrac{1}{2}$ з чого випливає щомає місце $\dfrac{-1+1}{3} = 0, 0 \in \mathbb{Z} $, а отже ми використаємо підстановку $a+bx^n = t^s$, де $s$ знаменник дробу $p$ .

$$
\Bigg|
  \begin{array}{lr rl rl rl}
    t^2 = & 1-x^3           & \d{x} =& (x)'\d{t}      & x_2 = & -\sqrt[3]{3}} & t_2 = & \sqrt{1-(-\sqrt[3]{3})^3} = 2  \\
    x   = & \sqrt[3]{1-t^2} & \d{x} =& - \dfrac{2t\d{t}}{3\sqrt[3]{(1-t^2)^2}}   & x_1 = & -2 & t_1 = & \sqrt{1-(-2)^3} = 3 & \\
    \end{array}
\Bigg|
$$

$$
= \int^2_3 - \dfrac{2t\d{t}}{\sqrt[3]{1-t^2} \cdot 3\sqrt[3]{(1-t^2)^2} }
= - \int^2_3 \dfrac{2t\d{t}}{3 \sqrt[3]{(1-t^2)^3} }
= - \dfrac{1}{3} \int^2_3 \dfrac{2t\d{t}}{1-t^2}
= - \dfrac{1}{3} \int^2_3 \dfrac{\d{(t^2-1)}}{-1(t^2-1)}
= \dfrac{1}{3} \int^2_3 \dfrac{\d{(t^2-1)}}{t^2-1}
$$
%
$$
= \dfrac{1}{3} \ln{t^2-1} \bigg|^2_3
= \dfrac{1}{3} ( \ln{|4-1|} - \ln{|9-1|} )
= \dfrac{1}{3} \ln{\dfrac{|4-1|}{|9-1|}}
= \dfrac{1}{3} \ln{\big|\dfrac{3}{8}\big|}
$$
