\M{\large{\textbf{Збіжність і Розбіжність}}}
\begin{multicols}{2}
  \M{Ряд \textbf{збігається} }

  $$ \lim_{x \to \infty}  S_n = S $$
  \M{Приклад ряду що \textbf{збігається} }

  $$ \boxed{ \sum_{n=0}^\infty aq^n = a+aq+aq^2 \ldots + aq^n , ( |q| < 1) } $$
  \M{\textit{Cпадаюча геометрична прогресія} }

  \M{Ряд \textbf{розбігається}}
  $$\lim_{x \to \infty}  S_n \neq S$$
  \M{Приклад ряду що \textbf{розбігається}  }

  $$\boxed{ \sum_{n=0}^\infty \dfrac{1}{n} = 1 + \dfrac{1}{2} + \dfrac{1}{3} \ldots + \dfrac{1}{n} }$$

  \M{\textit{Гармонічний ряд (завжди розбіжний)} }
\end{multicols}
