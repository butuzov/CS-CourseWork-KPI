\M{\large{\textbf{Ознака збіжності Даламбера}}}

$$
{\text{ Для числового ряду }}
\sum_{n=0}^{\infty} a
\qquad
{\text{ існує границя вигляду }}
\qquad
\lim_{n \to \infty} \dfrac{a_{n+1}}{a_n} = c
\begin{cases}
  c < 1    & {\text{ ряд збігається }} \\
  c > 1    & {\text{ ряд розбігається}} \\
\end{cases}
$$
