\M{\large{\textbf{Ознака збіжності Коші}}}

$$
{\text{ Для числового ряду }}
\sum_{n=0}^{\infty} a
\qquad
{\text{ існує границя вигляду }}
\qquad
\lim_{n \to \infty} \Sqrt[n]{a} = c
\begin{cases}
  c > 1    & {\text{ ряд розбігається}} \\
  c < 1    & {\text{ ряд збігається }} \\
  c \neq 1 & {\text{ не відомо }} \\
\end{cases}
$$
