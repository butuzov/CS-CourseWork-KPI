\M{\large{\textbf{Ряди}}}

\M{\textbf{Геометрична прогресія}}
\begin{multicols}{2}
  \M{\normalsize{\textbf{Збіжний ряд ( |q| < 1 )}}}
  $$\boxed{Sn= \sum_{n=0}^{\infty} q^n = \dfrac{a_1}{1-q}}$$
  \M{\normalsize{\textbf{Розбіжний ряд (|q| > 1)}}}
  $$\boxed{Sn= a\sum_{n=0}^{k} q^n = a_1 \dfrac{1-q^{n}}{1-q}}$$
\end{multicols}


\M{\textbf{Ряд Діріхле}}
\begin{multicols}{2}
  \M{\normalsize{\textbf{Збіжний ряд ( |q| > 1 )}}}
  $$\boxed{Sn= \sum_{n=0}^{\infty} \dfrac{1}{n^\alpha} }$$
  \M{\normalsize{\textbf{Розбіжний ряд (|q| $\leqslant$ 1)}}}
  $$\boxed{Sn= a\sum_{n=0}^{k} \dfrac{1}{n^\alpha}}$$
\end{multicols}
